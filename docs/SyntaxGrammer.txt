# this is the grammer defined for LOX language


Expression -> Ternary ;
Ternary -> Equality ( "?" Equality ":" Ternary )* ;
Equality -> Comparison (("!=" | "==" ) Comparison)* ;
Comparison -> Term ( (">=" | ">" | "<" | "<=" ) Term)* ;
Term -> Factor ( ("+" | "-") Factor)* ;
Factor -> Unary ( ("*" | "/") Unary)* ;
Unary -> ("!" | "-") Unary | Primary ;
Primary -> NUMBER | STRING | "true" | "false" | "nil"| "(" Expression ")"
